Shortcuts:

a -- Area
  - units: m2, a

d -- Data
  - units: b, B

e -- Energy
  - units: J, eV, cal, Wh

f -- Frequency
  - units: Hz, rpm

o -- Force
  - units: N, kgf, dyn

l -- Length
  - units: m, in, ft, yd, ly, au, pc, mi

n -- Number Systems
  - units: any natural number >= 2

w -- Power
  - units: W, hp, cal/h

p -- Pressure
  - units: Pa, atm, bar, mmHg

s -- Speed
  - units: length/time

t -- Temperature
  - units: K, C, F

i -- Time
  - units: s, min, h, d, w, y, dec, cent

v -- Volume
  - units: m3, L, gal, qt, tsp, tbsp, vk

m -- Weight and Mass
  - units: g, t, oz, lb
